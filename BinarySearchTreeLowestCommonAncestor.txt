import java.util.*;

class Node {
    Node left;
    Node right;
    int data;
    
    Node(int data) {
        this.data = data;
        left = null;
        right = null;
    }
}

	public static Node lca(Node root, int v1, int v2) {
      	while (root != null) {
        if (root.data > v1 && root.data > v2) {
            root = root.left;
        } else if (root.data < v1 && root.data < v2) {
            root = root.right;
        } else {
            break;
        } 
    }
    return root;
    }
